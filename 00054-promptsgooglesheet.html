  • Structure Google Sheet : Le script doPost génère automatiquement 9 colonnes (ID, Thème, Artiste, Prompt +, Prompt -, Seed, Lien Image, Note, Date) et les onglets par catégorie.

    • Accès via Extensions > Apps Script.
    • Déploiement via Gérer les déploiements > Nouvelle version (pour conserver l'URL /exec stable).
  • Google Colab :
    • Outil de transmission : colab.google.

LE SCRIPT (Google Sheet - APPS SCRIPT - déploiement) : [fichier "Code.gs"]
function doPost(e) { var ss = SpreadsheetApp.getActiveSpreadsheet(); var data = JSON.parse(e.postData.contents); var category = data.category ? data.category : "Général"; var sheet = ss.getSheetByName(category); // Création automatique de l'onglet si inexistant if (!sheet) { sheet = ss.insertSheet(category); var headers = ["ID", "Thème", "Artiste/Style", "Prompt Positif", "Prompt Négatif", "Seed", "Lien Image", "Note", "Date"]; sheet.appendRow(headers); sheet.getRange(1, 1, 1, 9).setBackground("#121212").setFontColor("white").setFontWeight("bold"); sheet.setFrozenRows(1); } // Calcul de l'ID (ex: #001) var id = "#" + (sheet.getLastRow()).toString().padStart(3, '0'); // 1. SÉCURITÉ ANNETTE : Verrou anti-signature systématique var posPrompt = (data.positive || "").trim() + ", Absolutely no signature"; // 2. VERROU NÉGATIF : Tag [NEG] + Protection contre le texte déformé var negPrompt = "[NEG] misspelling, distorted text, blurry letters, " + (data.negative || "").trim(); // Ajout de la ligne dans le tableau sheet.appendRow([ id, data.theme, data.artist, posPrompt, negPrompt, data.seed || "Auto", data.link || "", "", new Date() ]); // Formatage visuel (Couleurs et alignement) var lastRow = sheet.getLastRow(); sheet.getRange(lastRow, 4).setBackground("#EBF5FB"); // Bleu pour le Positif sheet.getRange(lastRow, 5).setBackground("#FDEDEC").setFontColor("#922B21"); // Rouge pour le Négatif sheet.getRange(lastRow, 1, 1, 9).setVerticalAlignment("top").setWrap(true); return ContentService.createTextOutput("Success : " + category).setMimeType(ContentService.MimeType.TEXT); }
