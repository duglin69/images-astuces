function doPost(e) {
  var sheet = SpreadsheetApp.getActiveSpreadsheet().getSheets()[0];
  var d = JSON.parse(e.postData.contents);
  
  sheet.appendRow([
    new Date(),      // A = DERNIER TEST À
    d.artistStyle,   // B = Artist_Style
    d.genre,         // C = Genre
    d.sujet,         // D = Sujet / Set-up
    d.tags,          // E = Tags / Ambiance
    d.promptFinal,   // F = Positive_Prompt
    d.negatif,       // G = Negative_Prompt
    d.ratio,         // H = Ratio
    d.seed,          // I = Seed
    d.link           // J = Link
  ]);
  return ContentService.createTextOutput("OK");
}

#@title ✨ STUDIO LUMIÈRE NOIRE - ALIGNEMENT COLONNES (A-J)
import ipywidgets as widgets
from IPython.display import display, HTML, clear_output
import requests

# --- INTERFACE ---
url_w = widgets.Text(description='🔗 URL /exec:', layout=widgets.Layout(width='95%'))
sujets_w = widgets.Textarea(description='✍️ Sujets (1 par ligne):', rows=6, layout=widgets.Layout(width='95%'))

# Tes sélections
artist_w = widgets.Dropdown(options=["SWEDISH SPIDER", "PHOTOGRAPHIC", "AUCUN"], description='B: Artist_Style', value="SWEDISH SPIDER")
genre_w = widgets.Dropdown(options=["CINEMATIC", "VINTAGE", "AUCUN"], description='C: Genre', value="AUCUN")
tags_w = widgets.Text(description='E: Tags/Ambiance', value="High Quality, 8k")
ratio_w = widgets.Dropdown(options=["16:9", "9:16", "1:1", "4:5"], description='H: Ratio', value="16:9")

btn = widgets.Button(description='🚀 ENVOYER AU SHEET', button_style='success', layout=widgets.Layout(width='95%', height='50px'))
out = widgets.Output()

def envoyer(b):
    with out:
        clear_output()
        url = url_w.value.strip()
        if not url:
            print("❌ Manque l'URL !"); return
            
        lignes = [l.strip() for l in sujets_w.value.split('\n') if l.strip()]
        for s in lignes:
            # Construction du Prompt Final (F)
            p_final = f"{s}, {artist_w.value}, {genre_w.value}, {tags_w.value} --ar {ratio_w.value}".strip()
            
            payload = {
                "artistStyle": artist_w.value,
                "genre": genre_w.value,
                "sujet": s,
                "tags": tags_w.value,
                "promptFinal": p_final,
                "negatif": "--no blur, low quality",
                "ratio": ratio_w.value,
                "seed": "", # Vide pour l'instant
                "link": ""  # Vide pour l'instant
            }
            
            try:
                r = requests.post(url, json=payload)
                print(f"✅ Envoyé dans Colonne B à J : {s[:20]}...")
            except Exception as e:
                print(f"❌ Erreur : {e}")

btn.on_click(envoyer)
display(url_w, sujets_w, artist_w, genre_w, tags_w, ratio_w, btn, out)
