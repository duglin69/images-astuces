📜 LE RÉCEPTIONNISTE (Google Apps Script)

À copier dans Extensions > Apps Script. Ce code crée les 10 colonnes si l'onglet est absent.

function doPost(e) {
  var data = JSON.parse(e.postData.contents);
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var sheet = ss.getSheetByName(data.sheetName);
  
  if (!sheet) {
    sheet = ss.insertSheet(data.sheetName);
    sheet.appendRow(["Format", "Artist_Style", "Genre", "Sujet / Set-up", "Tags / Ambiance", "Positive_Prompt", "Negative_Prompt", "Ratio", "Seed", "Link"]);
  }

  sheet.appendRow([
    data.Format, data.Artist_Style, data.Genre, data.Sujet, data.Tags, 
    data.positive, data.negative, data.Ratio, data.seed, data.link
  ]);
  
  return ContentService.createTextOutput("Injection réussie dans " + data.sheetName);
}

📜 LE TRANSPORTEUR (Python pour Colab)

Script à utiliser pour envoyer vos batchs de 50 prompts vers le tableau.

import requests
import json

URL_SHEET = "VOTRE_URL_DÉPLOYÉE_ICI/exec"

data_to_send = [
    {
        "sheetName": "MASTER_ART_BIBLE",
        "Format": "Illustration",
        "Artist_Style": "Comic Noir N&B",
        "Genre": "Polar",
        "Sujet": "Détective",
        "Tags": "Pluie, Ville sombre",
        "positive": "A noir and white comic illustration...",
        "negative": "color, blurry",
        "Ratio": "1:1",
        "seed": "Auto",
        "link": ""
    }
]

for item in data_to_send:
    response = requests.post(URL_SHEET, data=json.dumps(item))
    print(f"Statut : {response.text}")
